// Assets/Scripts/Chat/ChatIdFormat.cs â€” single home for id logic
public static class ChatIdFormat
{
    // Strip "@c.us" ONLY when the suffix is present (suffix-conditional, NOT channel-conditional).
    // Null/empty pass through unchanged. Groups (@g.us) and bare numeric TG ids keep their id.
    public static string Recipient(string id);

    // Name fallback when a dialog has no name. Strip @c.us / @g.us when present; otherwise
    // return the id VERBATIM. NEVER slice a numeric or short id, NEVER throw on empty.
    // Replaces ChatManager.cs:288 `chat.id[..^5]`.
    public static string DisplayFallback(string id);

    // Suffix-only overload (used by per-bubble message-view checks that only have a chatId).
    public static bool IsGroup(string chatId);
    // Full overload (used by the chat list, which has the dialog). WhatsApp: @g.us suffix OR
    // dialogIsGroup. Telegram: dialogType == "chat". Routes through the suffix check.
    public static bool IsGroup(string chatId, string dialogType, bool dialogIsGroup);
}
