        [SerializeField] private TextMeshProUGUI replyText;    // 42u, body #111B21, Ellipsis, 2-line cap

        public void Setup(SuggestionItem item, bool isTop)
        {
            replyText.text = item.text;
            intentLabel.text = item.intentLabel;
            recommendedBadge.SetActive(isTop);                 // index 0 ONLY
            cardButton.onClick.RemoveAllListeners();

                transform.DOPunchScale(Vector3.one * -0.03f, 0.15f, 0, 0).SetEase(Ease.OutQuad); // 0.97 punch, UI-SPEC

            });
        }
    }
